# FPYTHON pode ser escrito com minusculas ou maiusculas

S -> ABREFP Fpython FECHAFP

Fpython -> Fpython FUNCAO
         | FUNCAO

Funcao -> FUNABRE WORD ABREP Conjunto FECHAP Corpofun FUNFECHA
        | FUNABRE WORD ABREP FECHAP Corpofun FUNFECHA

Conjunto -> Var
         | Conjunto VIR Var

Corpofun -> RETURN Result PV 
          | IF ABREP Cond FECHAP Corpofun ELSE Corpofun

Cond -> Varoper Conds Varoper
      | NOT Cond
      | Cond CONDAND Cond
      | Cond CONDOR Cond
      | ABREP Cond FECHAP

Conds -> MENOR
       | MAIOR
       | IGUAL
       | DIFERENTE
       | MAIORIGUAL
       | MENORIGUAL
       | IN

Result -> Varoper
        | Opern Varoper
        | Result Oper Result
        | ABREP Result FECHAP

Opern -> ADD
       | MINUS

Oper -> Opern
      | TIMES
      | DIVIDE
      | MOD
      | OR
      | AND
      | Conds

Varoper -> Var
         | WORD ABREP Conjunto FECHAP
         | WORD ABREP FECHAP

Var -> NUMBER
     | BOOL
     | List
     | WORD

List -> ABREL FECHAL
      | ABREL Conjunto FECHAL
      | ABREL Conjunto2 FECHAL

Cojunto2 -> Cojunto2 NEXT WORD
          | WORD
